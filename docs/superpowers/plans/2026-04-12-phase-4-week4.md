```python
    """A -> B -> C, A -> D, T1 tests A, T2 tests B."""
    g = Graph()
    for src, dst in [
        ("a.py", "sym_a"), ("b.py", "sym_b"), ("c.py", "sym_c"), ("d.py", "sym_d"),

        g.add_relation(Relation(src_type="file", src_ref=src, dst_type="symbol",

    for src, dst in [
        ("b.py", "sym_a"), ("c.py", "sym_b"), ("d.py", "sym_a"),
    ]:
        g.add_relation(Relation(src_type="file", src_ref=src, dst_type="symbol",
```

```python
"""Per-file static impact analysis using dependency graph."""
from __future__ import annotations
from collections import deque
from dataclasses import dataclass, field
from libs.graph.builder import Graph

@dataclass(frozen=True)
class ImpactReport:
    target: str
    direct_dependents: list[str] = field(default_factory=list)
    transitive_dependents: list[str] = field(default_factory=list)
    affected_tests: list[str] = field(default_factory=list)
    risk_score: float = 0.0

def analyze_impact(
    target: str,
    graph: Graph,
    *,
    file_roles: dict[str, str] | None = None,
    max_depth: int = 4,
    git_churn: int = 0,
) -> ImpactReport:
    roles = file_roles or {}
    # Direct: files that import symbols defined by target
    direct = set()
    # Get symbols defined by target
    target_symbols = graph.neighbors(target)  # file -> symbols it defines/uses
    # Reverse: who depends on target
    rev_neighbors = graph.reverse_neighbors(target)
    # Filter to files only (not symbols)
    for dep in rev_neighbors:
        if "." not in dep or dep.endswith(".py") or dep.endswith(".j2"):
            direct.add(dep)

    # BFS for transitive
    visited: set[str] = {target}
    frontier: deque[tuple[str, int]] = deque()
    for d in direct:
        frontier.append((d, 1))
        visited.add(d)

    transitive: set[str] = set()
    while frontier:
        node, depth = frontier.popleft()
        transitive.add(node)
        if depth >= max_depth:
            continue
        for nxt in graph.reverse_neighbors(node):
            if nxt not in visited and (nxt.endswith(".py") or nxt.endswith(".j2")):
                visited.add(nxt)
                frontier.append((nxt, depth + 1))

    affected_tests = sorted(f for f in transitive if roles.get(f) == "test")
    fan_in = len(direct)
    fan_out = len(graph.neighbors(target))
    risk = fan_in * max(1, fan_out) * (1 + git_churn / 10)

    return ImpactReport(
        target=target,
        direct_dependents=sorted(direct),
        transitive_dependents=sorted(transitive - direct),
        affected_tests=affected_tests,
        risk_score=risk,
    )
```

```python
        file_degrees={"a.py": (5, 2), "b.py": (5, 2)},
        git_churn={"a.py": 3, "b.py": 3},
        test_coverage={"a.py": False, "b.py": True},
```

```python
"""Hotspot scoring: high fan_in × churn × no-tests = risk."""
from __future__ import annotations
from dataclasses import dataclass

@dataclass(frozen=True)
class HotspotEntry:
    file_path: str
    fan_in: int
    fan_out: int
    churn_30d: int
    has_tests: bool
    hotspot_score: float

def compute_hotspots(
    *,
    file_degrees: dict[str, tuple[int, int]],  # {path: (fan_in, fan_out)}
    git_churn: dict[str, int],
    test_coverage: dict[str, bool],
    limit: int = 10,
) -> list[HotspotEntry]:
    entries: list[HotspotEntry] = []
    for fp, (fan_in, fan_out) in file_degrees.items():
        churn = git_churn.get(fp, 0)
        has_tests = test_coverage.get(fp, False)
        score = fan_in * (1 + churn) * (2.0 if not has_tests else 1.0)
        entries.append(HotspotEntry(
            file_path=fp, fan_in=fan_in, fan_out=fan_out,
            churn_30d=churn, has_tests=has_tests, hotspot_score=score,
        ))
    entries.sort(key=lambda e: -e.hotspot_score)
    return entries[:limit]
```
